```go
package main

import (
	"fmt"
)

func main() {
	var n, c1index, c2index int
	n = 20

	a := make([]int, n)
	// second := make([]int, 2)

	for i := 0; i < n; i++ {
		fmt.Scan(&a[i])
	}

	fmt.Println(a)

	firstMaximum, secondMaximum := -1<<31, -1<<31 

	for i := 0; i < n; i++ {
		if a[i] > firstMaximum {
			secondMaximum = firstMaximum
			c2index = c1index
			firstMaximum = a[i]
			c1index = i
		} else if a[i] > secondMaximum && firstMaximum != a[i] {
			secondMaximum = a[i]
			c2index = i
		}
	}

	if c2index > c1index {
		fmt.Println(firstMaximum, secondMaximum)
	} else {
		fmt.Println(secondMaximum, firstMaximum)
	}
}
```

Пояснения:

- Цикл проходит по всем элементам массива.

- Если текущий элемент больше текущего максимума:

- Старый максимум становится вторым максимумом (secondMaximum = firstMaximum).

- Индекс второго максимума обновляется на старый индекс максимума (c2index = c1index).

- Новый максимум и его индекс обновляются (firstMaximum = a[i], c1index = i).

- Если элемент не больше максимума, но больше второго максимума и не равен максимуму:

- Второй максимум обновляется на этот элемент (secondMaximum = a[i]).

- Индекс второго максимума обновляется (c2index = i).
