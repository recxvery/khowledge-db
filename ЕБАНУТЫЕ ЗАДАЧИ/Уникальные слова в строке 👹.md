```go
package main

import (
	"bufio"
	"fmt"
	"os"
	"slices"
	"strings"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	wordInput, _ := reader.ReadString('\n')
	wordInput = strings.TrimSpace(wordInput)

	unsortedWords := strings.Split(wordInput, " ")

	words := make([]string, 0)
	for _, v := range unsortedWords {
		if !slices.Contains(words, v) {
			words = append(words, v)
		}
	}

	for i := 0; i < len(words) - 1; i++{
		swapped := false
		for j := 0; j < len(words) - 1; j++ {
			if words[j] > words[j+1] {
				words[j], words[j+1] = words[j+1], words[j]
				swapped = true
			} 
		if !swapped {
			swapped = false
		}
		}
	}

	for _, v := range words {
		fmt.Println(v)
	}
}
```
