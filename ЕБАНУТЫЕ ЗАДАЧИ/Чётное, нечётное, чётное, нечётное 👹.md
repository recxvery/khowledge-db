package main

import (
	"fmt"
)

func main() {
	var mincount, c int
	originalArray := make([]int, 15)
	countableArray := make([]int, 0)
	uncountableArray := make([]int, 0)

	for i := 0; i < 15; i++ {
		fmt.Scan(&originalArray[i])
	}

	for i := 0; i < 15; i++ {
		if originalArray[i]%2 == 0 {
			countableArray = append(countableArray, originalArray[i])
		} else {
			uncountableArray = append(uncountableArray, originalArray[i])
		}
	}


	
	if len(countableArray) > len(uncountableArray) {
		mincount = len(uncountableArray)
		c = 1
	} else {
		mincount = len(countableArray)
		c = 2
	}

	for j := 0; j < mincount; j++ {
		fmt.Printf("%v ", countableArray[j])
		fmt.Printf("%v ", uncountableArray[j])
	}
	if c == 1 {
		for v := mincount; v < len(countableArray); v++ {
			fmt.Printf("%v ", countableArray[v])
			} 
	} else {
		for v := mincount; v < len(uncountableArray); v++ {
			fmt.Printf("%v ", uncountableArray[v])
			} 
		}
}
