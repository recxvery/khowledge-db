Код примерно такой как ниже:

```cpp
class Student
{
	std::string m_first = "First";
	std::string m_last = "Second";
	int			m_id   = 0;
	float		m_avg  = 3.8f;

public:

	Student () {}
	//НИЖЕ Конструктор с параметрами(ахуенна увоуоу)
	Student(std::string first, std::string last, int id, float avg) //
		: m_first(first)// 
		, m_last(last)
		, m_id(id) //
		, m_avg(avg) // всё это мы прописываем на будущее для функции воид
	{
	}
	int getAvg()
	{
		return m_avg;
	}
	
	int getID()
	{
		return m_id;
	}
	std::string getFirst()
	{
		return m_first;
	}
	std::string getLast()
	{
		return m_last;
	}
```

Функция void:

```cpp

	void print() const // обязательно нужно прописать это чтобы в дальнешйем не было ошибки
	//при выводе значения
	{
		std::cout << m_first + ' ' + m_last << ' ';
		std::cout << m_id << " " << m_avg << "\n";
	} 
```

 По сути её используют сугубо чтобы печатать, то есть она никак не должна влиять на нашу переменную. просто выводить её на экран к примеру.

Дальнейший вывод:

```cpp

int main(int argc, char * argv[])
{
	Student s1;
	Student s2("Mikhail", "Trofimov", 1, 3.6);
	const Student s3("Agatha", "Kathagelophobia", 22002020220, 100.0);

	s3.print();

	return 0;
}

```
