func minSubArrayLen(target int, nums []int) int {
	minLen := math.MaxInt32
	left := 0
	sum := 0

	for right := 0; right < len(nums); right++ {
        sum += nums[right]
        for sum >= target {
            currentLen := right - left + 1
            minLen = min(minLen, currentLen)
            sum -= nums[left]
            left++
        }
	}

    if minLen == math.MaxInt32 {
        return 0
    }

	return minLen
}
